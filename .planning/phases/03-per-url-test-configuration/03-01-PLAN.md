    ```python
    """
    Per-URL check configuration for WatchDog.

    Loads config/url_checks.yaml and provides CheckConfig.enabled_checks_for(url)
    to filter ValidationResult objects by check type for a given URL.
    """
    import logging
    import yaml
    from typing import Dict, List
    from pydantic import BaseModel, field_validator
    from constants import KNOWN_CHECK_TYPES


    class UrlCheckSpec(BaseModel):
        enabled: List[str]

        @field_validator("enabled")
        @classmethod
        def _warn_unknown_check_names(cls, v: List[str]) -> List[str]:
            unknown = set(v) - KNOWN_CHECK_TYPES
            if unknown:
                logging.warning(
                    "Unknown check names in config (will never match any result): %s", unknown
                )
            return v


    class CheckConfig(BaseModel):
        version: int = 1
        defaults: UrlCheckSpec
        urls: Dict[str, UrlCheckSpec] = {}

        def enabled_checks_for(self, url: str) -> frozenset:
            """Return the set of enabled check type strings for a given URL.

            Normalizes trailing slashes on both the lookup URL and config keys
            so that 'https://allen.in/' and 'https://allen.in' resolve identically.
            Falls back to defaults.enabled when the URL is not explicitly configured.
            """
            normalized = url.rstrip("/")
            for config_url, spec in self.urls.items():
                if config_url.rstrip("/") == normalized:
                    return frozenset(spec.enabled)
            return frozenset(self.defaults.enabled)

        @classmethod
        def load(cls, path: str = "config/url_checks.yaml") -> "CheckConfig":
            """Load CheckConfig from a YAML file.

            Returns a permissive default config (all known checks enabled for all URLs)
            if the file does not exist, so existing deployments without a config file
            continue to behave exactly as before.
            """
            try:
                with open(path, encoding="utf-8") as fh:
                    data = yaml.safe_load(fh)
                return cls.model_validate(data)
            except FileNotFoundError:
                logging.warning(
                    "Check config %s not found — running all checks for all URLs", path
                )
                return cls(defaults=UrlCheckSpec(enabled=list(KNOWN_CHECK_TYPES)))
    ```

    KEY IMPLEMENTATION NOTES:
    - Use `yaml.safe_load()` — NEVER `yaml.load()` without an explicit Loader
    - Use `frozenset` return type for `enabled_checks_for()` — immutable, safe for set membership tests
    - The `@field_validator` must log a WARNING (not raise) so future check types added before KNOWN_CHECK_TYPES is updated don't break existing configs
    - Do NOT add `if not v: logging.info(...)` — empty list is valid and intentional; the caller (validate_course) handles empty set by returning no results

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Filesystem → Python process | `config/url_checks.yaml` is read from disk at runtime |
| Config data → Pydantic model | Raw YAML dict is validated against CheckConfig schema |
| Config keys → check filter | Check type strings from YAML are compared against ValidationResult.type via set membership |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Tampering | `yaml.safe_load()` in CheckConfig.load() | mitigate | Use `yaml.safe_load()` exclusively — never `yaml.load()` without Loader. `safe_load()` disallows arbitrary Python object construction (e.g. `!!python/object` tags). Enforced by acceptance criteria grep check. |
| T-03-02 | Elevation of Privilege | `CheckConfig.load(path)` path parameter | accept | Path is passed as a hardcoded string literal in ScraperEngine ("config/url_checks.yaml") — not user-supplied input. No path traversal risk in the call site. |
| T-03-03 | Tampering | Unknown check name silently disabling a check | mitigate | `@field_validator("enabled")` in UrlCheckSpec logs WARNING for any name not in KNOWN_CHECK_TYPES. Operator is alerted to typos at load time without crashing. Mitigates silent check bypass via misconfigured YAML. |
| T-03-04 | Information Disclosure | config/url_checks.yaml committed to git | accept | File contains only URL paths and check type names — no credentials, tokens, or PII. Safe to commit. |
</threat_model>
